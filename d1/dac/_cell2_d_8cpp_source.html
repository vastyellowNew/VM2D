111 void Cell::CreateAllCells(PointsCopy::iterator& itBegin, PointsCopy::iterator& itEnd)
112 {
113  double w = rightUp[0] - leftDown[0];
114  double h = rightUp[1] - leftDown[1];
115 
116  if (std::distance(itBegin, itEnd) > 1)
117  {
118  if (w > h)
119  {
120  double midX = (rightUp[0] + leftDown[0]) / 2.0;
121 
122  mChildren[0].reset(new Cell(tree, leftDown, { midX, rightUp[1] }));
123  mChildren[1].reset(new Cell(tree, { midX, leftDown[0] }, rightUp));
124 
125  // сортировка по х
126  std::sort(itBegin, itEnd, PointsCopy_iterator::LessXcc);
127 
128  PointsCopy::iterator itDop(itBegin);
129 
130  mChildren[0]->itBegin = itBegin;
131  mChildren[1]->itEnd = itEnd;
132 
133  bool flag = true;
134 
135  while (flag)
136  {
137  if (itDop.getVtx().r()[0] < midX)
138  itDop++;
139  else
140  {
141  flag = false;
142  mChildren[0]->itEnd = itDop;
143  mChildren[1]->itBegin = itDop;
144 
145  //обрезаем по вихрям
146  mChildren[0]->rightUp[0] = (itDop - 1).getVtx().r()[0];
147  mChildren[1]->leftDown[0] = itDop.getVtx().r()[0];
148  }
149  }//while (flag)
150 
151  // для границ по горизонтали
152  for (int i = 0; i < 2; ++i)
153  {
154  auto spanY = std::minmax_element(mChildren[i]->itBegin, mChildren[i]->itEnd, PointsCopy_iterator::LessYrr);
155 
156  mChildren[i]->leftDown[1] = spanY.first.getVtx().r()[1];
157  mChildren[i]->rightUp[1] = spanY.second.getVtx().r()[1];
158  }// for i
159  } //if (w > h)
160  else
161  {
162  double midY = (leftDown[1] + rightUp[1]) / 2.0;
163 
164  mChildren[0].reset(new Cell(tree, leftDown, { rightUp[0], midY }));
165  mChildren[1].reset(new Cell(tree, { leftDown[0], midY }, rightUp));
166 
167  //сортировка по у
168  std::sort(itBegin, itEnd, PointsCopy_iterator::LessYcc);
169 
170  PointsCopy::iterator itDop = itBegin;
171 
172  mChildren[0]->itBegin = itBegin;
173  mChildren[1]->itEnd = itEnd;
174 
175  bool flag = true;
176 
177  while (flag)
178  {
179  if (itDop.getVtx().r()[1] < midY)
180  itDop++;
181  else
182  {
183  flag = false;
184  mChildren[0]->itEnd = itDop;
185  mChildren[1]->itBegin = itDop;
186 
187  //обрезаем по вихрям
188  mChildren[0]->rightUp[1] = (itDop - 1).getVtx().r()[1];
189  mChildren[1]->leftDown[1] = itDop.getVtx().r()[1];
190  }
191  }//while (flag)
192 
193 
194  //для границ по вертикали
195  for (int i = 0; i < 2; ++i)
196  {
197  auto spanX = std::minmax_element(mChildren[i]->itBegin, mChildren[i]->itEnd, PointsCopy_iterator::LessXrr);
198 
199  mChildren[i]->leftDown[0] = spanX.first.getVtx().r()[0];
200  mChildren[i]->rightUp[0] = spanX.second.getVtx().r()[0];
201  }
202  } //else
203 
204  // номер уровня потомков
205  for (int j = 0; j < 2; ++j)
206  {
207  mChildren[j]->level = level + 1;
208 
209  if ((mChildren[j]->level < tree.numOfLevels) && ((mChildren[j]->itEnd - mChildren[j]->itBegin) > tree.minNumOfVort) && \
210  (std::max(fabs(mChildren[j]->rightUp[0] - mChildren[j]->leftDown[0]), fabs(mChildren[j]->rightUp[1] - mChildren[j]->leftDown[1])) > tree.minDist))
211  mChildren[j]->CreateAllCells(mChildren[j]->itBegin, mChildren[j]->itEnd);
212  else
213  {
214  mChildren[j]->lowLevel = true;
215  tree.lowCells.push_back(mChildren[j].get());
216  }
217  }
218  }
219  else
220  {
221  lowLevel = true;
222  tree.lowCells.push_back(this);
223  }
224 }//CreateAllCells(...)

226 //Вычисление параметров всех ячеек дерева (циркуляций и центров завихренности)
227 void Cell::CalculateCellsParams()
228 {
229  double gg[2];
230  Point2D rr[2];
231 
232  for (int i = 0; i < 2; ++i)
233  {
234  gg[i] = 0.0;
235  rr[i].toZero();
236  }
237 
238  if (!lowLevel)
239  {
240  mChildren[0]->CalculateCellsParams();
241  mChildren[1]->CalculateCellsParams();
242 
243  for (int k = 0; k < 2; ++k)
244  {
245  gg[k] = mChildren[0]->sumGam[k] + mChildren[1]->sumGam[k];
246 
247  if (fabs(gg[k]) > 1e-10) //когда все вихри внутри области одной циркуляции
248  {
249  rr[k] += mChildren[0]->sumGam[k] * mChildren[0]->centerGam[k] + mChildren[1]->sumGam[k] * mChildren[1]->centerGam[k];
250  rr[k] /= gg[k];
251  }
252  sumGam[k] = gg[k];
253  centerGam[k] = rr[k];
254  centerGam[k] = rr[k];
255  }// for k
256  }//if(!lowLevel)
257 
258  else
259  {
260  for(auto it = itBegin; it != itEnd; ++it)
261  {
262  double gDop = it.getVtx().g();
263  int k = (int)(gDop < 0.0);
264  gg[k] += gDop;
265  rr[k] += gDop * it.getVtx().r();
266  }
267 
268  for (int k = 0; k < 2; ++k)
269  {
270  if (fabs(gg[k]) > 1e-10)
271  rr[k] /= gg[k];
272 
273  sumGam[k] = gg[k];
274  centerGam[k] = rr[k];
275  }// for k
276  }//else
277 }//CalculateCellsParams()

323 //Расчет коэффициентов a,b,c,d для ячейки нижнего уровня
324 void Cell::CalcABCDandCloseCellsToLowLevel(Cell& infCell, bool calcCoef)
325 {
326  if (&infCell != this)
327  {
328  //сумма габаритов ячейки, на которую считается влияние (h0), и влияющей (h)
329  double h0, h;
330  h0 = fabs(rightUp[0] - leftDown[0]) + fabs(rightUp[1] - leftDown[1]);
331  h = fabs(infCell.rightUp[0] - infCell.leftDown[0]) + fabs(infCell.rightUp[1] - infCell.leftDown[1]);
332 
333  //r0 --- нижний уровень, r1 --- влияющий, p --- с плюсом, m --- с минусом
334  Point2D r0, r1, r1p, r1m;
335 
336  // центры прямоугольников
337  r0 = 0.5 * (rightUp + leftDown);
338  r1 = 0.5 * (infCell.rightUp + infCell.leftDown);
339 
340  double crit = dist(r0, r1);
341  //double crit = sqrt(sqr(r0[0] - r1[0]) + sqr(r0[1] - r1[1]));
342 
343  // если выполнен критерий дальности => считаем коэффициенты
344  if ((crit >= (h0 + h + tree.W.getPassport().wakeDiscretizationProperties.eps) / tree.theta))
345  {
346  if (calcCoef)
347  {
348  if ((itEnd - itBegin) <= 3)
349  VMlib::ModifyE2(closestCellDist.data(), crit);
350 
351  //центры завихренностей
352  r1p = infCell.centerGam[0];
353  r1m = infCell.centerGam[1];
354 
355  //суммарные циркуляции
356  double gam[2];
357  gam[0] = infCell.sumGam[0];
358  gam[1] = infCell.sumGam[1];
359 
360  Point2D rrP, rrM;
361  rrP = r0 - r1p;
362  rrM = r0 - r1m;
363 
364  double nrmP2 = rrP.length2();
365  double nrmM2 = rrM.length2();
366 
367  a += -IDPI * (gam[0] * rrP[1] / nrmP2 + gam[1] * rrM[1] / nrmM2);
368  b += IDPI * (gam[0] * rrP[0] / nrmP2 + gam[1] * rrM[0] / nrmM2);
369 
370  double nrmP4 = nrmP2 * nrmP2;
371  double nrmM4 = nrmM2 * nrmM2;
372 
373  c += 2.0 * IDPI * (gam[0] * rrP[0] * rrP[1] / nrmP4 + \
374  gam[1] * rrM[0] * rrM[1] / nrmM4);
375 
376  d += IDPI * (gam[0] * (rrP[1] * rrP[1] - rrP[0] * rrP[0]) / nrmP4 + \
377  gam[1] * (rrM[1] * rrM[1] - rrM[0] * rrM[0]) / nrmM4);
378  }
379  }//if crit
380  else // если не выполнен критерий, то рекурсия
381  {
382  if (!(infCell.lowLevel))
383  {
384  CalcABCDandCloseCellsToLowLevel(*(infCell.mChildren[0]), calcCoef);
385  CalcABCDandCloseCellsToLowLevel(*(infCell.mChildren[1]), calcCoef);
386  }
387  else
388  closeCells.push_back(&infCell);
389  }
390  }//if (lowTree != this)
391  else closeCells.push_back(&infCell); //себя тоже добавляем в ближнюю зону
392 
393 }//CalcABCDandCloseCellsToLowLevel(...)

535 void Cell::CalcConvVeloByBiotSavartFromSources()
