49 // Функция-множитель, позволяющая моделировать разгон
50 double PhysicalProperties::accelCft() const
51 {
52  switch (typeAccel.second)
53  {
54  case 0: //импульсный старт
55  return 1.0;
56  case 1: //разгон потока по линейному закону
57  return (timeProp.currTime < timeAccel) ? (timeProp.currTime / timeAccel) : 1.0;
58  case 2: //разгон потока по косинусоиде
59  return (timeProp.currTime < timeAccel) ? 0.5 * (1.0 - cos(PI * timeProp.currTime / timeAccel)) : 1.0;
60  }
61 
62  return 1.0;
63 }//accelCft()

105 //Считывание всех параметров расчета из соответствующих потоков
106 void Passport::GetAllParamsFromParser
107 (
108  std::istream& mainStream,
109  std::istream& mechanicsStream,
110  std::istream& defaultStream,
111  std::istream& switcherStream,
112  std::istream& varsStream
113 )
114 {
115 
116 // 1. Разбор паспорта в целом
117 
118  //создаем парсер и связываем его с нужным потоком
119  std::unique_ptr<VMlib::StreamParser> parser;
120  parser.reset(new VMlib::StreamParser(info, "parser", mainStream, defaultStream, switcherStream, varsStream));
121 
122  //считываем общие парамеры
123  parser->get("rho", physicalProperties.rho);
124  parser->get("vInf", physicalProperties.vInf);
125  parser->get("vRef", physicalProperties.vRef, &defaults::defaultVRef);
126 
127  if (physicalProperties.vRef == 0.0)
128  {
129  if (physicalProperties.vInf.length() == 0.0)
130  {
131  info('e') << "Reference velocity should be non-zero!" << std::endl;
132  exit(1);
133  }
134  physicalProperties.vRef = physicalProperties.vInf.length();
135  }
136 
137  //Считывание схемы разгона потока
138  std::pair<std::pair<std::string, int>, std::string> velAccel;
139  bool defParam = parser->get("accelVel", velAccel, &defaults::defaultVelAccel);
140  if (velAccel.first.second != -1)
141  {
142  physicalProperties.typeAccel = velAccel.first;
143  std::stringstream sstr(velAccel.second);
144  std::istream& istr(sstr);
145 
146  std::unique_ptr<VMlib::StreamParser> parserAccel;
147 
148  parserAccel.reset(new VMlib::StreamParser(info, "parserAccel", istr, defaultStream, switcherStream, varsStream));
149  bool defParamTime = parserAccel->get("timeAccel", physicalProperties.timeAccel, &defaults::defaultTimeAccel, false);
150  parserAccel->get("_DEFVAR_0", physicalProperties.timeAccel, &physicalProperties.timeAccel, defParam && !defParamTime);
151  }
152  else
153  {
154  info('e') << "Velocity acceleration scheme <" << velAccel.first.first << "> is unknown" << std::endl;
155  exit(1);
156  }
157 
159  if (!defParam)
160  parser->get("timeAccel", physicalProperties.timeAccel, &defaults::defaultTimeAccel);
161  else
162  {
163  double tempTimeAccel;
164  if (parser->get("timeAccel", tempTimeAccel, &defaults::defaultTimeAccel, false))
165  {
166  info('e') << "timeAccel parameter is set twice!" << std::endl;
167  exit(1);
168  }
169  }
170 
171 
172  parser->get("nu", physicalProperties.nu);
173 
174  parser->get("timeStart", timeDiscretizationProperties.timeStart, &defaults::defaultTimeStart);
175  parser->get("timeStop", timeDiscretizationProperties.timeStop);
176  parser->get("dt", timeDiscretizationProperties.dt);
177  parser->get("nameLength", timeDiscretizationProperties.nameLength, &defaults::defaultNameLength);
178  parser->get("saveVTK", timeDiscretizationProperties.saveVTK, &defaults::defaultSaveVTK);
179  parser->get("saveVP", timeDiscretizationProperties.saveVP, &defaults::defaultSaveVP);
180 
181 
182  parser->get("eps", wakeDiscretizationProperties.eps);
183  wakeDiscretizationProperties.eps2 = wakeDiscretizationProperties.eps* wakeDiscretizationProperties.eps;
184  parser->get("epscol", wakeDiscretizationProperties.epscol);
185  parser->get("distFar", wakeDiscretizationProperties.distFar, &defaults::defaultDistFar);
186  parser->get("delta", wakeDiscretizationProperties.delta, &defaults::defaultDelta);
187  parser->get("vortexPerPanel", wakeDiscretizationProperties.minVortexPerPanel, &defaults::defaultVortexPerPanel);
188  parser->get("maxGamma", wakeDiscretizationProperties.maxGamma, &defaults::defaultMaxGamma);
189  if (wakeDiscretizationProperties.maxGamma == 0.0)
190  wakeDiscretizationProperties.maxGamma = 1e+10;
191 
192  parser->get("linearSystemSolver", numericalSchemes.linearSystemSolver, &defaults::defaultLinearSystemSolver);
193  parser->get("velocityComputation", numericalSchemes.velocityComputation, &defaults::defaultVelocityComputation);
194  //parser->get("wakeMotionIntegrator", numericalSchemes.wakeMotionIntegrator);
195  parser->get("panelsType", numericalSchemes.panelsType, &defaults::defaultPanelsType);
196  parser->get("boundaryConditionSatisfaction", numericalSchemes.boundaryCondition, &defaults::defaultBoundaryCondition);
197 
198  parser->get("airfoilsDir", airfoilsDir, &defaults::defaultAirfoilsDir);
199  parser->get("wakesDir", wakesDir, &defaults::defaultWakesDir);
200 
201  parser->get("fileWake", wakeDiscretizationProperties.fileWake, &defaults::defaultFileWake);
202  parser->get("fileSource", wakeDiscretizationProperties.fileSource, &defaults::defaultFileSource);
203 
204  std::vector<std::string> airfoil;
205  parser->get("airfoil", airfoil, &defaults::defaultAirfoil);
206 
207  // 2. Разбор параметров профилей
208 
209  //определяем число профилей и организуем цикл по ним
210  size_t nAirfoil = airfoil.size();
211  //*(defaults::defaultPinfo) << "Number of airfoils = " << nAirfoil << endl;
212  for (size_t i = 0; i < nAirfoil; ++i)
213  {
214  //делим имя файла + выражение в скобках на 2 подстроки
215  std::pair<std::string, std::string> airfoilLine = VMlib::StreamParser::SplitString(info, airfoil[i], false);
216 
217  AirfoilParams prm;
218  //первая подстрока - имя файла
219  prm.fileAirfoil = airfoilLine.first;
220 
221  //вторую подстроку разделяем на вектор из строк по запятым, стоящим вне фигурных скобок
222  std::vector<std::string> vecAirfoilLineSecond = VMlib::StreamParser::StringToVector(airfoilLine.second, '{', '}');
223 
224  //создаем парсер и связываем его с параметрами профиля
225  std::stringstream aflStream(VMlib::StreamParser::VectorStringToString(vecAirfoilLineSecond));
226  std::unique_ptr<VMlib::StreamParser> parserAirfoil;
227 
228  parserAirfoil.reset(new VMlib::StreamParser(info, "airfoil parser", aflStream, defaultStream, switcherStream, varsStream));
229 
230  //считываем нужные параметры с учетом default-значений
231  parserAirfoil->get("basePoint", prm.basePoint, &defaults::defaultBasePoint);
232  parserAirfoil->get("scale", prm.scale, &defaults::defaultScale);
233  parserAirfoil->get("angle", prm.angle, &defaults::defaultAngle);
234  prm.angle *= PI / 180.0;
235 
236  parserAirfoil->get("inverse", prm.inverse, &defaults::defaultInverse);
237  parserAirfoil->get("mechanicalSystem", prm.mechanicalSystem, &defaults::defaultMechanicalSystem);
238 
239  if (prm.mechanicalSystem == defaults::defaultMechanicalSystem)
240  {
241  prm.mechanicalSystemType = 0;
242  prm.mechanicalSystemParameters = "";
243  }
244  else
245  {
246  std::unique_ptr<VMlib::StreamParser> parserMechanicsList;
247  std::unique_ptr<VMlib::StreamParser> parserSwitchers;
248  parserMechanicsList.reset(new VMlib::StreamParser(info, "mechanical parser", mechanicsStream, defaultStream, switcherStream, varsStream, { prm.mechanicalSystem }));
249  parserSwitchers.reset(new VMlib::StreamParser(info, "switchers parser" ,switcherStream));
250 
251  std::string mechString;
252 
253  parserMechanicsList->get(prm.mechanicalSystem, mechString);
254 
255  //делим тип мех.системы + выражение в скобках (ее параметры) на 2 подстроки
256  std::pair<std::string, std::string> mechanicsLine = VMlib::StreamParser::SplitString(info, mechString);
257 
258  std::string mechTypeAlias = mechanicsLine.first;
259  parserSwitchers->get(mechTypeAlias, prm.mechanicalSystemType);
260 
261  //вторую подстроку разделяем на вектор из строк по запятым, стоящим вне фигурных скобок
262  std::vector<std::string> vecMechLineSecond = VMlib::StreamParser::StringToVector(mechanicsLine.second, '{', '}');
263  prm.mechanicalSystemParameters = VMlib::StreamParser::VectorStringToString(vecMechLineSecond);
264  }
265 
266  //отправляем считанные параметры профиля в структуру данных паспорта
267  airfoilParams.push_back(prm);
268 
269  } //for i
270 }//GetAllParamsFromParser(...)
