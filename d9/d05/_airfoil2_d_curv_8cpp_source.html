195 void AirfoilCurv::GetGabarits(double gap) //определение габаритного прямоугольника
196 {
197  lowLeft = { 1E+10, 1E+10 };
198  upRight = { -1E+10, -1E+10 };
199 
200  for (size_t i = 0; i < r_.size(); ++i)
201  {
202  lowLeft[0] = std::min(lowLeft[0], r_[i][0]);
203  lowLeft[1] = std::min(lowLeft[1], r_[i][1]);
204 
205  upRight[0] = std::max(upRight[0], r_[i][0]);
206  upRight[1] = std::max(upRight[1], r_[i][1]);
207  }
208 
209  for (size_t i = 0; i < r_.size(); ++i)
210  {
211  lowLeft[0] = std::min(lowLeft[0], rc_[i][0]);
212  lowLeft[1] = std::min(lowLeft[1], rc_[i][1]);
213 
214  upRight[0] = std::max(upRight[0], rc_[i][0]);
215  upRight[1] = std::max(upRight[1], rc_[i][1]);
216  }
217 
218  Point2D size = upRight - lowLeft;
219  lowLeft -= size*gap;
220  upRight += size*gap;
221 }//GetGabarits(...)
