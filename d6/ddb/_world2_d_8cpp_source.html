367 void World2D::FillMatrixAndRhs()
368 {
369  getTimestat().timeFillMatrixAndRhs.first += omp_get_wtime();
370 
371  Eigen::MatrixXd locMatr;
372  Eigen::MatrixXd otherMatr;
373  Eigen::VectorXd locLastLine, locLastCol;
374 
375  std::vector<std::vector<Point2D>> locIQ;
376  std::vector<std::vector<Point2D>> locOtherIQ;
377 
378  //обнуляем матрицу на первом шаге расчета
379  if (currentStep == 0)
380  {
381  for (int i = 0; i < matr.rows(); ++i)
382  for (int j = 0; j < matr.cols(); ++j)
383  matr(i, j) = 0.0;
384  }
385 
386  size_t currentRow = 0;
387 
388  for (size_t bou = 0; bou < boundary.size(); ++bou)
389  {
390  size_t nVars;
391 
392  if (parallel.myidWork == 0)
393  {
394  nVars = boundary[bou]->GetUnknownsSize();
395 
396  if (currentStep == 0)
397  {
398  locMatr.resize(nVars, nVars);
399  locLastLine.resize(nVars);
400  locLastCol.resize(nVars);
401  }
402  }
403 
404  if (currentStep == 0 || mechanics[bou]->isDeform)
405  {
406  if (parallel.myidWork == 0)
407  {
408  boundary[bou]->FillMatrixSelf(locMatr, locLastLine, locLastCol);
409  }
410  }
411 
412  //размазываем матрицу
413  if (parallel.myidWork == 0)
414  {
415  for (size_t i = 0; i < nVars; ++i)
416  {
417  if (currentStep == 0 || mechanics[bou]->isDeform)
418  {
419  for (size_t j = 0; j < nVars; ++j)
420  matr(i + currentRow, j + currentRow) = locMatr(i, j);
421  matr(currentRow + nVars, i + currentRow) = locLastLine(i);
422  matr(i + currentRow, currentRow + nVars) = locLastCol(i);
423  }
424  }
425 
427  if (currentStep == 0)
428  {
429  size_t currentCol = 0;
430  for (size_t oth = 0; oth < boundary.size(); ++oth)
431  {
432  size_t nVarsOther = boundary[oth]->GetUnknownsSize();
433 
434  if (bou != oth)
435  {
436  otherMatr.resize(nVars, nVarsOther);
437 
438  boundary[bou]->FillMatrixFromOther(*boundary[oth], otherMatr);
439 
440  //размазываем матрицу
441  for (size_t i = 0; i < nVars; ++i)
442  {
443  for (size_t j = 0; j < nVarsOther; ++j)
444  matr(i + currentRow, j + currentCol) = otherMatr(i, j);
445  }
446  }// if (bou != oth)
447  currentCol += nVarsOther + 1;
448  }// for oth
449  }// if (currentStep == 0 || mechanics[oth]->isMoves)
450 
451  currentRow += nVars + 1;
452 
453  }// if (parallel.myidWork == 0)
454  }// for bou
455 
456  velocity->FillRhs(rhs);
457 
458  getTimestat().timeFillMatrixAndRhs.second += omp_get_wtime();
459 }//FillMatrixAndRhs()

503 void World2D::BuildTree(const PointType type) const
504 {
505 
506  //const int nLevelWake = 14;
507  //const int nLevelVP = 14;
508  //const int nLevelSorces = 5;
509  //const int nLevelSheets = 8;
510 
511  const int nLevelWake = 14;
512  const int nLevelVP = 1;
513  const int nLevelSources = 1;
514  const int nLevelSheets = 10;
515 
516 
517 
518  switch (type)
519  {
520  case PointType::wake:
521  {
522  if (wake->vtx.size() > 0)
523  treeWake.reset(new Tree(*this, *wake, PointType::wake, nLevelWake));
524  else
525  treeWake = nullptr;
526  break;
527  }
528  case PointType::wakeVP:
529  {
530  if (measureVP->getWakeVP().vtx.size() > 0)
531  treeVP.reset(new Tree(*this, measureVP->getWakeVP(), PointType::wakeVP, nLevelVP));
532  else
533  treeVP = nullptr;
534  break;
535  }
536  case PointType::sourceWake:
537  {
538  if (source->vtx.size() > 0)
539  treeSourcesWake.reset(new Tree(*this, *source, PointType::sourceWake, nLevelSources));
540  else
541  treeSourcesWake = nullptr;
542  break;
543  }
544 
545  case PointType::sheetGam:
546  {
547  if (boundary.size() > 0)
548  treeSheetsGam.reset(new Tree(*this, PointType::sheetGam, nLevelSheets));
549  else
550  treeSheetsGam = nullptr;
551  break;
552  }
553  case PointType::source:
554  {
555  if (boundary.size() > 0)
556  treeSheetsSource.reset(new Tree(*this, PointType::source, nLevelSheets));
557  else
558  treeSheetsSource = nullptr;
559  break;
560  }
561 
562  default:
563  break;
564  }
