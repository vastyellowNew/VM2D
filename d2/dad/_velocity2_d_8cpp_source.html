141 #endif
142  }
143 
144  //Порядок циклов именно такой, т.к. CUDA оптимизирует вызов ядер,
145  //и на один "слой" виртуальных вихрей считается влияние сразу от всех профилей
146  for (size_t bou = 0; bou < W.getNumberOfBoundary(); ++bou)
147  {
148  for (size_t afl = 0; afl < W.getNumberOfAirfoil(); ++afl)
149 #if (defined(__CUDACC__) || defined(USE_CUDA)) && (defined(CU_I0I3))
150  W.getNonConstAirfoil(afl).GPUGetDiffVelocityI0I3ToSetOfPointsAndViscousStresses(W.getBoundary(bou).virtualWake, virtualVortexesParams[bou].epsastWake, virtualVortexesParams[bou].I0, virtualVortexesParams[bou].I3);
151 #else
152  W.getNonConstAirfoil(afl).GetDiffVelocityI0I3ToSetOfPointsAndViscousStresses(W.getBoundary(bou).virtualWake, virtualVortexesParams[bou].epsastWake, virtualVortexesParams[bou].I0, virtualVortexesParams[bou].I3);
153 #endif
154  }
155  //влияние поверхности
156  Point2D I3;
157  double I0;
158 
159  double domrad = 0.0;
160 
161  for (size_t vt = 0; vt < wakeVortexesParams.diffVelo.size(); ++vt)
162  {
163  domrad = std::max(wakeVortexesParams.epsastWake[vt], W.getPassport().wakeDiscretizationProperties.getMinEpsAst());
164 
165  wakeVortexesParams.I0[vt] *= domrad;
166  wakeVortexesParams.I0[vt] += DPI * sqr(domrad);
167 
168  I3 = wakeVortexesParams.I3[vt];
169  I0 = wakeVortexesParams.I0[vt];
170 
171  if (fabs(I0) > 1.e-8)
172  wakeVortexesParams.diffVelo[vt] += I3 * (1.0 / I0);
173  }
174 
175  for (size_t targetBou = 0; targetBou < W.getNumberOfBoundary(); ++targetBou)
176  for (size_t vt = 0; vt < virtualVortexesParams[targetBou].diffVelo.size(); ++vt)
177  {
178  domrad = std::max(virtualVortexesParams[targetBou].epsastWake[vt], W.getPassport().wakeDiscretizationProperties.getMinEpsAst());
179 
180  virtualVortexesParams[targetBou].I0[vt] *= domrad;
181  virtualVortexesParams[targetBou].I0[vt] += DPI * sqr(domrad);
182 
183  I3 = virtualVortexesParams[targetBou].I3[vt];
184  I0 = virtualVortexesParams[targetBou].I0[vt];
185 
186  if (fabs(I0) > 1.e-8)
187  virtualVortexesParams[targetBou].diffVelo[vt] += I3 * (1.0 / I0);
188  }
189 }//CalcDiffVeloI0I3()

261 void Velocity::CalcDiffVeloI1I2ToSetOfPointsFromWake(const WakeDataBase& pointsDb, const std::vector<double>& domainRadius, const WakeDataBase& vorticesDb, std::vector<double>& I1, std::vector<Point2D>& I2)
262 {
263  double tCPUSTART, tCPUEND;
264 
265  tCPUSTART = omp_get_wtime();
266 
267  std::vector<double> selfI1;
268  std::vector<Point2D> selfI2;
269 
270  const int& id = W.getParallel().myidWork;
271 
272  VMlib::parProp par = W.getParallel().SplitMPI(pointsDb.vtx.size());
273 
274  std::vector<Vortex2D> locPoints;
275  locPoints.resize(par.myLen);
276 
277  MPI_Scatterv(const_cast<std::vector<Vortex2D/*, VM2D::MyAlloc<VMlib::Vortex2D>*/>&>(pointsDb.vtx).data(), par.len.data(), par.disp.data(), Vortex2D::mpiVortex2D, \
278  locPoints.data(), par.myLen, Vortex2D::mpiVortex2D, 0, W.getParallel().commWork);
279 
280  std::vector<double> locI1(par.myLen, 0.0);
281  std::vector<Point2D> locI2(par.myLen, { 0.0, 0.0 });
282 
283 #pragma warning (push)
284 #pragma warning (disable: 4101)
285  //Локальные переменные для цикла
286  Point2D Rij;
287  double rij, expr;
288  double diffRadius, domRad;
289  double left;
290  double right;
291  double posJx;
292 #pragma warning (pop)
293 
294 #pragma omp parallel for default(none) shared(locI1, locI2, domainRadius, locPoints, vorticesDb, par) private(Rij, rij, expr, diffRadius, domRad, left, right, posJx)
295  for (int i = 0; i < par.myLen; ++i)
296  {
297  const Vortex2D& vtxI = locPoints[i];
298 
299  domRad = std::max(domainRadius[i + par.myDisp], W.getPassport().wakeDiscretizationProperties.getMinEpsAst());
300 
302  diffRadius = 8.0 * domRad;
303 
304  left = vtxI.r()[0] - diffRadius;
305  right = vtxI.r()[0] + diffRadius;
306 
307  for (size_t j = 0; j < vorticesDb.vtx.size(); ++j)
308  {
309  const Vortex2D& vtxJ = vorticesDb.vtx[j];
310  posJx = vtxJ.r()[0];
311 
312  if ((left < posJx) && (posJx < right))
313  {
314  Rij = vtxI.r() - vtxJ.r();
315  rij = Rij.length();
316  if (rij < diffRadius && rij > 1.e-10)
317  {
318  expr = exp(-rij / domRad);
319  locI2[i] += (vtxJ.g()* expr / rij) * Rij;
320  locI1[i] += vtxJ.g()*expr;
321  }
322  }//if (rij>1e-6)
323  }//for j
324  } // for r
325 
326  if (id == 0)
327  {
328  selfI1.resize(pointsDb.vtx.size(), 0.0);
329  selfI2.resize(pointsDb.vtx.size(), { 0.0, 0.0 });
330  }
331 
332  MPI_Gatherv(locI1.data(), par.myLen, MPI_DOUBLE, selfI1.data(), par.len.data(), par.disp.data(), MPI_DOUBLE, 0, W.getParallel().commWork);
333 
334  MPI_Gatherv(locI2.data(), par.myLen, Point2D::mpiPoint2D, selfI2.data(), par.len.data(), par.disp.data(), Point2D::mpiPoint2D, 0, W.getParallel().commWork);
335 
336  if (id == 0)
337  for (size_t i = 0; i < I1.size(); ++i)
338  {
339  I1[i] += selfI1[i];
340  I2[i] += selfI2[i];
341  }
342 
343  tCPUEND = omp_get_wtime();
344  //W.getInfo('t') << "DIFF_CPU: " << tCPUEND - tCPUSTART << std::endl;
345 }//CalcDiffVeloI1I2ToSetOfPointsFromWake(...)
