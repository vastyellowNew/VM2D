248 void VelocityBarnesHut::GetWakeInfluenceToRhs(std::vector<double>& wakeRhs) const
249 {
250  Tree& treeWakeRef = W.getNonConstTree(PointType::wake);
251  Tree& treeSheetsGamRef = W.getNonConstTree(PointType::sheetGam);
252  Tree& treeSheetsSourceRef = W.getNonConstTree(PointType::source);
253  //Tree& treeSourceWakeRef = W.getNonConstTree(PointType::sourceWake);
254 
255  //Здесь предполагается, что тип схемы у всех профилей одинаковый
256  size_t shDim = W.getBoundary(0).sheetDim;
257  std::vector<double> velI(shDim, 0.0);
258 
259  Point2D approxVel;
260  for (size_t i = 0; i < treeSheetsGamRef.lowCells.size(); ++i)
261  {
262 
263  if (W.treeSheetsSource)
264  {
265  treeSheetsGamRef.lowCells[i]->ClearABCD();
266  treeSheetsGamRef.lowCells[i]->closeCells.clear();
267  treeSheetsGamRef.lowCells[i]->CalcABCDandCloseCellsToLowLevel(treeSheetsSourceRef.rootCell, true);
268 
269  for (auto it = treeSheetsGamRef.lowCells[i]->itBegin; it != treeSheetsGamRef.lowCells[i]->itEnd; ++it) // цикл по панелям
270  {
271  int aflN = it.getAflPnl().first;
272  int panNum = it.getAflPnl().second;
273 
274  velI.assign(shDim, 0.0);
275 
276  for (size_t j = 0; j < treeSheetsGamRef.lowCells[i]->closeCells.size(); ++j)
277  W.getAirfoil(aflN).GetInfluenceFromSourcesToPanel(panNum, &(treeSheetsGamRef.lowCells[i]->closeCells[j]->itBegin.getVtx()), \
278  std::distance(treeSheetsGamRef.lowCells[i]->closeCells[j]->itBegin, treeSheetsGamRef.lowCells[i]->closeCells[j]->itEnd), velI);
279 
280  for (size_t j = 0; j < shDim; ++j)
281  velI[j] *= IDPI / W.getAirfoil(aflN).len[panNum];
282 
283  // результат приближенного влияния записывается в veloCopy конкретной панели (там уже есть накопленные влияния от дальних источников)
284  treeSheetsGamRef.lowCells[i]->CalcInfluenceFromSourceFarCells(it);
285 
286  velI[0] += treeSheetsGamRef.allPnt.velo[it.getNum()] & W.getAirfoil(aflN).tau[panNum];
287 
288  for (size_t k = 0; k < shDim; ++k)
289  wakeRhs[W.getDispBoundaryInSystem(aflN) + panNum + k * W.getAirfoil(aflN).getNumberOfPanels()] = velI[k];
290  }
291  }//if(treeSources)
292 
293  treeSheetsGamRef.lowCells[i]->ClearABCD();
294  treeSheetsGamRef.lowCells[i]->closeCells.clear();
295 
296  // аналогично для wake
297  if (W.treeWake)
298  {
299  treeSheetsGamRef.lowCells[i]->CalcABCDandCloseCellsToLowLevel(treeWakeRef.rootCell, true);
300  //int count = 0;
301  for (auto it = treeSheetsGamRef.lowCells[i]->itBegin; it != treeSheetsGamRef.lowCells[i]->itEnd; ++it) // цикл по панелям
302  {
303  //if (count == 0)
304  //std::cout << treeSheetsGam->lowCells[i]->itEnd - treeSheetsGam->lowCells[i]->itBegin << std::endl;
305  int aflN = it.getAflPnl().first;
306  int panNum = it.getAflPnl().second;
307 
308  velI.assign(shDim, 0.0);
309 
310  for (size_t j = 0; j < treeSheetsGamRef.lowCells[i]->closeCells.size(); ++j)
311  W.getAirfoil(aflN).GetInfluenceFromVorticesToPanel(panNum, &(treeSheetsGamRef.lowCells[i]->closeCells[j]->itBegin.getVtx()), \
312  std::distance(treeSheetsGamRef.lowCells[i]->closeCells[j]->itBegin, treeSheetsGamRef.lowCells[i]->closeCells[j]->itEnd), velI);
313 
314  for (size_t j = 0; j < shDim; ++j)
315  velI[j] *= IDPI / W.getAirfoil(aflN).len[panNum];
316 
317  // результат приближенного влияния записывается в velo конкретной панели (там уже есть накопленные влияния от дальних источников)
318  treeSheetsGamRef.lowCells[i]->CalcInfluenceFromVortexFarCells(it);
319 
320  velI[0] += treeSheetsGamRef.allPnt.velo[it.getNum()] & W.getAirfoil(aflN).tau[panNum];
321 
322  for (size_t k = 0; k < shDim; ++k)
323  wakeRhs[W.getDispBoundaryInSystem(aflN) + panNum + k * W.getAirfoil(aflN).getNumberOfPanels()] += velI[k];
324 
325  //std::cout << count << std::endl;
326  //count++;
327  }
328  }//if (treeWake)
329  }
330 
331 }//GetWakeInfluenceToRhs(...)

394 void VelocityBarnesHut::GetWakeInfluenceToRhsBS(const Airfoil& afl, std::vector<double>& wakeRhs) const
395 {
396  size_t np = afl.getNumberOfPanels();
397  int id = W.getParallel().myidWork;
398  VMlib::parProp par = W.getParallel().SplitMPI(np);
399 
400  std::vector<double> locVeloWake, locVeloWakeLin;
401  locVeloWake.resize(par.myLen);
402 
403  size_t shDim = W.getBoundary(afl.numberInPassport).sheetDim;
404 
405  if (shDim != 1)
406  locVeloWakeLin.resize(par.myLen);
407 
408  //локальные переменные для цикла
409  std::vector<double> velI(shDim, 0.0);
410 
411 #pragma omp parallel for default(none) shared(locVeloWake, par, shDim, afl, locVeloWakeLin, IDPI) private(velI)
412  for (int i = 0; i < par.myLen; ++i)
413  {
414  velI.assign(shDim, 0.0);
415 
416  if (W.getWake().vtx.size() > 0)
417  {
418  //Учет влияния следа
419  afl.GetInfluenceFromVorticesToPanel(par.myDisp + i, &(*W.getWake().vtx.begin()), std::distance(W.getWake().vtx.begin(), W.getWake().vtx.end()), velI);
420  }
421 
422  if (W.getSource().vtx.size() > 0)
423  {
424  //Учет влияния источников
425  afl.GetInfluenceFromSourcesToPanel(par.myDisp + i, &(*W.getSource().vtx.begin()), std::distance(W.getSource().vtx.begin(), W.getSource().vtx.end()), velI);
426  }
427 
428  for (size_t j = 0; j < shDim; ++j)
429  velI[j] *= IDPI / afl.len[par.myDisp + i];
430 
431  locVeloWake[i] = velI[0];
432 
433  if (shDim != 1)
434  locVeloWakeLin[i] = velI[1];
435  }
436 
437  if (id == 0)
438  wakeRhs.resize(W.getBoundary(afl.numberInPassport).GetUnknownsSize());
439 
440  MPI_Gatherv(locVeloWake.data(), par.myLen, MPI_DOUBLE, wakeRhs.data(), par.len.data(), par.disp.data(), MPI_DOUBLE, 0, W.getParallel().commWork);
441 
442  if (shDim != 1)
443  MPI_Gatherv(locVeloWakeLin.data(), par.myLen, MPI_DOUBLE, wakeRhs.data() + np, par.len.data(), par.disp.data(), MPI_DOUBLE, 0, W.getParallel().commWork);
444 }//GetWakeInfluenceToRhsBS(...)

446 void VelocityBarnesHut::FillRhs(Eigen::VectorXd& rhs) const
447 {
448  Eigen::VectorXd locRhs;
449  std::vector<double> lastRhs(W.getNumberOfBoundary());
450 
451  size_t currentRow = 0;
452 
453  for (size_t bou = 0; bou < W.getNumberOfBoundary(); ++bou)
454  {
455  //double tt0 = omp_get_wtime();
456 
457  const Airfoil& afl = W.getAirfoil(bou);
458  size_t np = afl.getNumberOfPanels();
459 
460  size_t nVars;
461 
462  if (W.getParallel().myidWork == 0)
463  {
464  nVars = W.getBoundary(bou).GetUnknownsSize();
465 
466  locRhs.resize(nVars);
467  }
468 
469 
470  std::vector<double> wakeRhs;
471 
472 
473 #if (defined(__CUDACC__) || defined(USE_CUDA)) && (defined(CU_RHS))
474  GPUGetWakeInfluenceToRhs(afl, wakeRhs);
475 #else
476  GetWakeInfluenceToRhsBS(afl, wakeRhs);
477 #endif
478 
479  std::vector<double> vInfRhs;
480  afl.GetInfluenceFromVInfToPanel(vInfRhs);
481 
482  if (W.getParallel().myidWork == 0)
483  {
484 #pragma omp parallel for \
485  default(none) \
486  shared(locRhs, afl, bou, wakeRhs, vInfRhs, np) \
487  schedule(dynamic, DYN_SCHEDULE)
488  for (int i = 0; i < (int)afl.getNumberOfPanels(); ++i)
489  {
491  //Здесь учет скорости профиля и присоединенных слоев источников сделан только для прямолинейных панелей
492  locRhs(i) = -vInfRhs[i] - wakeRhs[i] + 0.25 * ((afl.getV(i) + afl.getV(i + 1)) & afl.tau[i]); //0.25 * (afl.getV(i) + afl.getV(i + 1))*afl.tau[i] - прямолинейные
493  if (W.getBoundary(bou).sheetDim > 1)
494  locRhs(np + i) = -vInfRhs[np + i] - wakeRhs[np + i];
495 
496  //влияние присоединенных слоев от самого себя и от других профилей
497  for (size_t q = 0; q < W.getNumberOfBoundary(); ++q)
498  {
499  const auto& sht = W.getBoundary(q).sheets;
500  const auto& iq = W.getIQ(bou, q);
501 
502  const Airfoil& aflOther = W.getAirfoil(q);
503  if (W.getMechanics(q).isDeform || W.getMechanics(q).isMoves)
504  {
505  for (size_t j = 0; j < aflOther.getNumberOfPanels(); ++j)
506  {
507  if ((i != j) || (bou != q))
508  {
