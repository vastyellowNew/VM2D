121 // Создание массива указателей на массив точек (для сортировки), используется для массива sheetsCopy
122 void Tree::CreateSheetsCopy(const PointType type)
123 {
124  size_t nSh = 0;
125  for (size_t i = 0; i < W.getNumberOfBoundary(); ++i)
126  nSh += W.getBoundary(i).sheets.getSheetSize();
127 
128  allPnt.reserve(nSh);
129  int num;
130  switch (type)
131  {
132  case PointType::sheetGam:
133  num = 0;
134  for (int i = 0; i < (int)(W.getNumberOfBoundary()); ++i)
135  for (int j = 0; j < (int)(W.getBoundary(i).sheets.getSheetSize()); ++j)
136  {
137  allPnt.emplace_back(Vortex2D{ 0.5 * (W.getAirfoil(i).getR(j) + W.getAirfoil(i).getR(j + 1)), W.getBoundary(i).sheets.attachedVortexSheet(j, 0) * W.getAirfoil(i).len[j] }, { i, j }, num, type);
138  num++;
139  }
140  break;
141  case PointType::source:
142  num = 0;
143  for (int i = 0; i < (int)(W.getNumberOfBoundary()); ++i)
144  for (int j = 0; j < (int)(W.getBoundary(i).sheets.getSheetSize()); ++j)
145  {
146  allPnt.emplace_back(Vortex2D{ 0.5 * (W.getAirfoil(i).getR(j) + W.getAirfoil(i).getR(j + 1)), W.getBoundary(i).sheets.attachedSourceSheet(j, 0) * W.getAirfoil(i).len[j] }, { i, j }, num, type);
147  num++;
148  }
149  break;
150  default:
151  break;
152  }
153 }//CreateSheetsCopy(...)
