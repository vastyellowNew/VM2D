56 //Вычисление гидродинамической силы, действующей на профиль
57 void MechanicsRigidOscillPart::GetHydroDynamForce()
58 {
59  W.getTimestat().timeGetHydroDynamForce.first += omp_get_wtime();
60 
61  const double& dt = W.getPassport().timeDiscretizationProperties.dt;
62 
63  hydroDynamForce = { 0.0, 0.0 };
64  hydroDynamMoment = 0.0;
65 
66  viscousForce = { 0.0, 0.0 };
67  viscousMoment = 0.0;
68 
69  Point2D hDFGam = { 0.0, 0.0 }; //гидродинамические силы, обусловленные Gamma_k
70  Point2D hDFdelta = { 0.0, 0.0 }; //гидродинамические силы, обусловленные delta_k
71  double hDMdelta = 0.0;
72 
73 
74  Point2D deltaVstep = { 0.0, u - uOld };
75 
76  for (size_t i = 0; i < afl.getNumberOfPanels(); ++i)
77  {
79  double deltaK = boundary.sheets.freeVortexSheet(i, 0) * afl.len[i] - afl.gammaThrough[i] + (deltaVstep & afl.tau[i]) * afl.len[i];
80  Point2D rK = 0.5 * (afl.getR(i + 1) + afl.getR(i)) - afl.rcm;
81 
82  hDFdelta += deltaK * Point2D({ -rK[1], rK[0] });
83  hDMdelta += 0.5 * deltaK * rK.length2();
84  }
85 
86  hydroDynamForce = hDFdelta * (1.0 / dt);
87  hydroDynamMoment = hDMdelta / dt;
88 
89  if (W.getPassport().physicalProperties.nu > 0.0)
90  for (size_t i = 0; i < afl.getNumberOfPanels(); ++i)
91  {
92  Point2D rK = 0.5 * (afl.getR(i + 1) + afl.getR(i)) - afl.rcm;
93  viscousForce += afl.viscousStress[i] * afl.tau[i];
94  viscousMoment += (afl.viscousStress[i] * afl.tau[i]) & rK;
95  }
96 
97  W.getTimestat().timeGetHydroDynamForce.second += omp_get_wtime();
98 }// GetHydroDynamForce()

122 void MechanicsRigidOscillPart::Move()
123 {
124  uOld = u;
125  yOld = y;
126 
127  double dy, du;
128 
129  //W.getInfo('t') << "k = " << k << std::endl;
130 
131  if (W.getParallel().myidWork == 0)
132  {
133  double dt = W.getPassport().timeDiscretizationProperties.dt;
134  numvector<double, 2> kk[4];
135  kk[0] = { u, (hydroDynamForce[1] - b*u - k*y) / m };
136  kk[1] = { u + 0.5*dt*kk[0][1], (hydroDynamForce[1] - b*(u + 0.5*dt*kk[0][1]) - k*(y + 0.5*dt*kk[0][0])) / m };
137  kk[2] = { u + 0.5*dt*kk[1][1], (hydroDynamForce[1] - b*(u + 0.5*dt*kk[1][1]) - k*(y + 0.5*dt*kk[1][0])) / m };
138  kk[3] = { u + dt*kk[2][1], (hydroDynamForce[1] - b*(u + dt*kk[2][1]) - k*(y + dt*kk[2][0])) / m };
139 
140  dy = dt * (kk[0][0] + 2. * kk[1][0] + 2. * kk[2][0] + kk[3][0]) / 6.0;
141  du = dt * (kk[0][1] + 2. * kk[1][1] + 2. * kk[2][1] + kk[3][1]) / 6.0;
142  }
143 
144  MPI_Bcast(&dy, 1, MPI_DOUBLE, 0, W.getParallel().commWork);
145  MPI_Bcast(&du, 1, MPI_DOUBLE, 0, W.getParallel().commWork);
146 
147  afl.Move({ 0.0, dy });
148  Vcm[1] += du;
149 
150  y += dy;
151  u += du;
152 }//Move()
