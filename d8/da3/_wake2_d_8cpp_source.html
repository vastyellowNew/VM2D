263 bool Wake::MoveInsideMovingBoundary(const Point2D& newPos, const Point2D& oldPos, const Airfoil& oldAfl, const Airfoil& afl, size_t& panThrough)
264 {
265  panThrough = -1;
266 
268 
269  //проверка габ. прямоугольника
270  if (afl.isOutsideGabarits(newPos) && afl.isOutsideGabarits(oldPos))
271  return false;
272 
273  bool hit = false;
274 
275  double angle = 0;
276  double cs, sn;
277 
278  double dist2 = 1000000000.0;
279 
280  for (size_t i = 0; i < afl.getNumberOfPanels(); ++i)
281  {
282  Point2D v1, v2, vv;
283  v1 = afl.getR(i) - newPos;
284 
285  v2 = afl.getR(i + 1) - newPos;
286 
287  vv = afl.getR(i + 1) - afl.getR(i);
288 
289  double dst = v1.length2() + v2.length2();
290  if (dst < dist2)
291  {
292  dist2 = dst;
293  panThrough = i;
294  }
295 
296  cs = v1 & v2;
297  sn = v1 ^ v2;
298 
299  angle += atan2(sn, cs);
300  }//for i
301 
302  hit = ((angle > 3.14) || (angle < -3.14));
303 
304  return hit;
305 }//MoveInsideMovingBoundary(...)

434 {
435  int id = W.getParallel().myidWork;

437 
438  std::vector<int> locNeighb; //локальный массив соседей (для данного процессора)
439  locNeighb.resize(par.myLen);
440 
441  Point2D Ri, Rk;
442 
443  double maxG = W.getPassport().wakeDiscretizationProperties.maxGamma;
444 
445 #pragma omp parallel for default(none) shared(type, locNeighb, par, maxG) schedule(dynamic, DYN_SCHEDULE)
446  for (int locI = 0; locI < par.myLen; ++locI)
447  {
448  int s = locI + par.myDisp;
449  const Vortex2D& vtxI = vtx[s];
450 
451  locNeighb[locI] = 0;//по умолчанию
452 
453  bool found = false;
454 
455  double r2, r2test;
456 
457  const double& cSP = collapseScaleParameter;
458  const double& cRBP = collapseRightBorderParameter;
459 
460  while ( (!found) && ( s + 1 < (int)vtx.size() ) )
461  {
462  s++;
463  const Vortex2D& vtxK = vtx[s];
464 
465  r2 = dist2(vtxI.r(), vtxK.r());
466 
467  //линейное увеличение радиуса коллапса
468  double mnog = std::max(1.0, (vtxI.r()[0] - cRBP) / cSP);
469 
470  r2test = sqr( W.getPassport().wakeDiscretizationProperties.epscol * mnog );
471 
472  if (type == 1)
473  r2test *= 4.0; //Увеличение радиуса коллапса в 2 раза для коллапса вихрей разных знаков
474 
475  if (r2 < r2test)
476  {
477  switch (type)
478  {
479  case 0:
480  found = ( fabs(vtxI.g()*vtxK.g()) != 0.0) && (fabs(vtxI.g() + vtxK.g()) < mnog * maxG);
481  break;
482  case 1:
483  found = (vtxI.g()*vtxK.g() < 0.0);
484  break;
485  case 2:
486  found = (vtxI.g()*vtxK.g() > 0.0) && (fabs(vtxI.g() + vtxK.g()) < mnog * maxG);
487  break;
488  }
489  }//if r2 < r2_test
490  }//while
491 
492  if (found)
493  locNeighb[locI] = s;
494  }//for locI
495 
496  if (id == 0)
497  neighb.resize(vtx.size());
498 
499  MPI_Gatherv(locNeighb.data(), par.myLen, MPI_INT, neighb.data(), par.len.data(), par.disp.data(), MPI_INT, 0, W.getParallel().commWork);
500 
501 }//GetPairsBS(...)
