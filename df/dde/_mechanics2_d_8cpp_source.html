147 void Mechanics::GenerateForcesString()
148 {
149  W.getTimestat().timeOther.first += omp_get_wtime();
150 
151  std::stringstream forceFileName, forceFileNameCsv;
152  forceFileName << W.getPassport().dir << "forces-airfoil-" << numberInPassport;
153  forceFileNameCsv << W.getPassport().dir << "forces-airfoil-" << numberInPassport << ".csv";
154 
155  double cShock = (W.getPassport().physicalProperties.getCurrTime() > W.getPassport().physicalProperties.timeAccel + 2.0 * W.getPassport().timeDiscretizationProperties.dt) ? 1.0 : 0.0;
156 
157  std::ofstream forcesFile(forceFileName.str(), std::ios::app);
158  forcesFile << std::endl << W.getCurrentStep() << " " << W.getPassport().physicalProperties.getCurrTime() << " " << cShock * hydroDynamForce[0] << " " << cShock * hydroDynamForce[1] << " " << cShock * hydroDynamMoment << " " << cShock * viscousForce[0] << " " << cShock * viscousForce[1] << " " << cShock * viscousMoment;
159  forcesFile.close();
160 
161  std::ofstream forcesFileCsv(forceFileNameCsv.str(), std::ios::app);
162  forcesFileCsv << std::endl << W.getPassport().physicalProperties.getCurrTime() << "," << cShock * hydroDynamForce[0] << "," << cShock * hydroDynamForce[1] << "," << cShock * hydroDynamMoment << "," << cShock * viscousForce[0] << "," << cShock * viscousForce[1] << "," << cShock * viscousMoment;
163  forcesFileCsv.close();
164 
165  W.getTimestat().timeOther.second += omp_get_wtime();
166 }//GenerateForcesString()
