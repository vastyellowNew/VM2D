127 void VelocityBiotSavart::CalcVeloToWakeVP()
128 {
129  std::vector<Point2D> velConvWake;
130  std::vector<std::vector<Point2D>> velConvBou;
131 
132  int addWSize = (int)W.getMeasureVP().getWakeVP().vtx.size();
133  MPI_Bcast(&addWSize, 1, MPI_INT, 0, W.getParallel().commWork);
134 
135  velConvWake.resize(addWSize, { 0.0, 0.0 });
136 
137  velConvBou.resize(W.getNumberOfBoundary());
138  for (size_t i = 0; i < W.getNumberOfBoundary(); ++i)
139  velConvBou[i].resize(addWSize, { 0.0, 0.0 });
140 
141 #if (defined(USE_CUDA))
142  W.getNonConstCuda().RefreshVP();
143 #endif
144 
145 #if (defined(__CUDACC__) || defined(USE_CUDA)) && (defined(CU_VP))
146  GPUCalcConvVeloToSetOfPointsFromWake(W.getMeasureVP().getWakeVP(), velConvWake, W.getNonConstMeasureVP().getNonConstDomainRadius(), true, false);
147 #else
148  CalcConvVeloToSetOfPointsFromWake(W.getMeasureVP().getWakeVP(), velConvWake, W.getNonConstMeasureVP().getNonConstDomainRadius(), true, false);
149 #endif
150 
151  for (size_t i = 0; i < W.getNumberOfBoundary(); ++i)
152 #if (defined(__CUDACC__) || defined(USE_CUDA)) && (defined(CU_VP))
153  W.getNonConstBoundary(i).GPUCalcConvVelocityToSetOfPointsFromSheets(W.getMeasureVP().getWakeVP(), velConvBou[i]);
154 #else
155  W.getNonConstBoundary(i).CalcConvVelocityToSetOfPointsFromSheets(W.getMeasureVP().getWakeVP(), velConvBou[i]);
156 #endif
157 
158  if (W.getParallel().myidWork == 0)
159  {
160  std::vector<Point2D>& velocityRef = W.getNonConstMeasureVP().getNonConstVelocity();
161  velocityRef.assign(addWSize, W.getPassport().physicalProperties.V0());
162 
163  for (int i = 0; i < addWSize; ++i)
164  velocityRef[i] += velConvWake[i];
165 
166  for (size_t bou = 0; bou < velConvBou.size(); ++bou)
167  for (int j = 0; j < addWSize; ++j)
168  velocityRef[j] += velConvBou[bou][j];
169  }
170 }// CalcVeloToWakeVP()

172 inline void ModifyE2(double* ee2, double dst2)
173 {
174  if (dst2 > 0)
175  {
176  if (dst2<ee2[0])
177  {
178  ee2[2] = ee2[1];
179  ee2[1] = ee2[0];
180  ee2[0] = dst2;
181  }//if (dist2<ee2[0])
182  else
183  {
184  if (dst2<ee2[1])
185  {
186  ee2[2] = ee2[1];
187  ee2[1] = dst2;
188  }// if (dist2<ee2[1])
189  else
190  if (dst2<ee2[2])
191  ee2[2] = dst2;
192  }//else
193  }//if (dst2>0)
194 }

196 //Вычисление конвективных скоростей и радиусов вихревых доменов в заданном наборе точек от следа
197 void VelocityBiotSavart::CalcConvVeloToSetOfPointsFromWake(const WakeDataBase& pointsDb, std::vector<Point2D>& velo, std::vector<double>& domainRadius, bool calcVelo, bool calcRadius)
198 {
199  std::vector<Point2D> selfVelo;
200 
201  const int& id = W.getParallel().myidWork;
202 
203  VMlib::parProp par = W.getParallel().SplitMPI(pointsDb.vtx.size(), true);
204 
205  std::vector<Vortex2D> locPoints;
206  locPoints.resize(par.myLen);
207 
208  //рассылка точек вычисления скоростей и радиусов доменов
209  MPI_Scatterv(const_cast<std::vector<Vortex2D/*, VM2D::MyAlloc<VMlib::Vortex2D>*/>&>(pointsDb.vtx).data(), par.len.data(), par.disp.data(), Vortex2D::mpiVortex2D, \
210  locPoints.data(), par.myLen, Vortex2D::mpiVortex2D, 0, W.getParallel().commWork);
211 
212  double cft = IDPI;
213 
214  std::vector<Point2D> locConvVelo;
215  if (calcVelo)
216  locConvVelo.resize(par.myLen);
217 
218  std::vector<double> locDomRadius;
219  if (calcRadius)
220  locDomRadius.resize(par.myLen);
221 
222 #pragma warning (push)
223 #pragma warning (disable: 4101)
224  //Локальные переменные для цикла
225  Point2D velI;
226  Point2D tempVel;
227  double dst2eps, dst2;
228 #pragma warning (pop)
229 
230  double eps2 = W.getPassport().wakeDiscretizationProperties.eps2;
231 
232  if (calcVelo)
233  {
234 #pragma omp parallel for default(none) shared(locConvVelo, locDomRadius, locPoints, cft, par, calcVelo, calcRadius, eps2) private(tempVel, velI, dst2, dst2eps) schedule(dynamic, DYN_SCHEDULE)
235  for (int i = 0; i < par.myLen; ++i)
236  {
237  double ee2[3] = { 10000.0, 10000.0, 10000.0 };
238 
239  velI.toZero();
240 
241  const Point2D& posI = locPoints[i].r();
242 
243  for (size_t j = 0; j < W.getWake().vtx.size(); ++j)
244  {
245  const Point2D& posJ = W.getWake().vtx[j].r();
246 
247  dst2 = dist2(posI, posJ);
248 
249  //Модифицируем массив квадратов расстояний до ближайших вихрей из wake
250 #ifndef TESTONLYVELO
251  if (calcRadius)
252  VMlib::ModifyE2(ee2, dst2);
253 #endif // !TESTONLYVELO
254 
255  const double& gamJ = W.getWake().vtx[j].g();
256 
257  tempVel.toZero();
258  dst2eps = VMlib::boundDenom(dst2, eps2); //Сглаживать надо!!!
259 
260  tempVel = { -posI[1] + posJ[1], posI[0] - posJ[0] };
261  tempVel *= (gamJ / dst2eps);
262  velI += tempVel;
263  }
264 
265  for (size_t j = 0; j < W.getSource().vtx.size(); ++j)
266  {
267  const Point2D& posJ = W.getSource().vtx[j].r();
268  const double& gamJ = W.getPassport().physicalProperties.accelCft() * W.getSource().vtx[j].g();
269 
270  tempVel.toZero();
271 
272  dst2 = dist2(posI, posJ);
273  dst2eps = VMlib::boundDenom(dst2, W.getPassport().wakeDiscretizationProperties.eps2); //Сглаживать надо!!!
274 
275  tempVel = { posI[0] - posJ[0], posI[1] - posJ[1] };
276  tempVel *= (gamJ / dst2eps);
277  velI += tempVel;
278  }
279 #ifndef TESTONLYVELO
280  if (calcRadius)
281  {
282  for (size_t s = 0; s < W.getNumberOfBoundary(); ++s)
283  {
284  const auto& bou = W.getBoundary(s);
285  //Модифицируем массив квадратов расстояний до ближайших вихрей из virtualWake
286  for (size_t j = 0; j < bou.virtualWake.vtx.size(); ++j)
287  ModifyE2(ee2, dist2(posI, bou.virtualWake.vtx[j].r()));
288  }
289  }
290 #endif
291 
292  velI *= cft;
293  locConvVelo[i] = velI;
294 
295 #ifndef TESTONLYVELO
296  if (calcRadius)
297  locDomRadius[i] = sqrt((ee2[0] + ee2[1] + ee2[2]) / 3.0);
298 #endif
299  }
300  }
301  else if (calcRadius)
302  {
303 #pragma omp parallel for default(none) shared(locConvVelo, locDomRadius, locPoints, cft, par, calcVelo, calcRadius) private(tempVel, velI, dst2) schedule(dynamic, DYN_SCHEDULE)
304  for (int i = 0; i < par.myLen; ++i)
305  {
306  double ee2[3] = { 10000.0, 10000.0, 10000.0 };
307 
308  velI.toZero();
309 
310  const Point2D& posI = locPoints[i].r();
311 
312  for (size_t j = 0; j < W.getWake().vtx.size(); ++j)
313  {
314  const Point2D& posJ = W.getWake().vtx[j].r();
315 
316  dst2 = dist2(posI, posJ);
317 
318  //Модифицируем массив квадратов расстояний до ближайших вихрей из wake
319  VMlib::ModifyE2(ee2, dst2);
320  }
321 
322  for (size_t s = 0; s < W.getNumberOfBoundary(); ++s)
323  {
324  for (size_t j = 0; j < W.getBoundary(s).virtualWake.vtx.size(); ++j)
325  {
326  const Point2D& posJ = W.getBoundary(s).virtualWake.vtx[j].r();
327  dst2 = dist2(posI, posJ);
328 
329  //Модифицируем массив квадратов расстояний до ближайших вихрей из virtualWake
330  ModifyE2(ee2, dst2);
331  }
332  }
333 
334  locDomRadius[i] = sqrt((ee2[0] + ee2[1] + ee2[2]) / 3.0);
335  }
336  } //else
337 
338 
339  if (calcVelo)
340  {
341  if (id == 0)
342  selfVelo.resize(pointsDb.vtx.size());
343  MPI_Gatherv(locConvVelo.data(), par.myLen, Point2D::mpiPoint2D, selfVelo.data(), par.len.data(), par.disp.data(), Point2D::mpiPoint2D, 0, W.getParallel().commWork);
344  }
345 
346 #ifndef TESTONLYVELO
347  if (calcRadius)
348  {
349  //if (id == 0)
350  //это нужно всем, т.к. ниже стоит Allgatherv
351  domainRadius.resize(par.totalLen);
352 
353  MPI_Allgatherv(locDomRadius.data(), par.myLen, MPI_DOUBLE, domainRadius.data(), par.len.data(), par.disp.data(), MPI_DOUBLE, W.getParallel().commWork);
354  }
355 #endif
356 
357  if ( (calcVelo) && (id == 0) )
358  for (size_t i = 0; i < velo.size(); ++i)
359  velo[i] += selfVelo[i];
360 }//CalcConvVeloToSetOfPointsFromWake(...)

490 void VelocityBiotSavart::GetWakeInfluenceToRhs(const Airfoil& afl, std::vector<double>& wakeRhs) const
491 {
492  size_t np = afl.getNumberOfPanels();
493  int id = W.getParallel().myidWork;
494  VMlib::parProp par = W.getParallel().SplitMPI(np);
495 
496  std::vector<double> locVeloWake, locVeloWakeLin;
497  locVeloWake.resize(par.myLen);
498 
499  size_t shDim = W.getBoundary(afl.numberInPassport).sheetDim;
500 
501  if(shDim != 1)
502  locVeloWakeLin.resize(par.myLen);
503 
504  //локальные переменные для цикла
505  std::vector<double> velI(shDim, 0.0);
506 
507 #pragma omp parallel for default(none) shared(locVeloWake, par, shDim, afl, locVeloWakeLin, IDPI) private(velI)
508  for (int i = 0; i < par.myLen; ++i)
509  {

511

513  {
514  //Учет влияния следа
515  afl.GetInfluenceFromVorticesToPanel(par.myDisp + i, &(*W.getWake().vtx.begin()), std::distance(W.getWake().vtx.begin(), W.getWake().vtx.end()), velI);
516  }
517 
518  if (W.getSource().vtx.size() > 0)

610 void VelocityBiotSavart::FillRhs(Eigen::VectorXd& rhs) const
611 {
612  Eigen::VectorXd locRhs;
613  std::vector<double> lastRhs(W.getNumberOfBoundary());
614 
615  size_t currentRow = 0;
616 
617  for (size_t bou = 0; bou < W.getNumberOfBoundary(); ++bou)
618  {
619  //double tt0 = omp_get_wtime();
620 
621  const Airfoil& afl = W.getAirfoil(bou);
622  size_t np = afl.getNumberOfPanels();
623 
624  size_t nVars;
625 
626  if (W.getParallel().myidWork == 0)
627  {
628  nVars = W.getBoundary(bou).GetUnknownsSize();
629 
630  locRhs.resize(nVars);
631  }
632 
633 
634  std::vector<double> wakeRhs;
635 
636 
637 #if (defined(__CUDACC__) || defined(USE_CUDA)) && (defined(CU_RHS))
638  GPUGetWakeInfluenceToRhs(afl, wakeRhs);
639 #else
640  GetWakeInfluenceToRhs(afl, wakeRhs);
641 #endif
642 
643  std::vector<double> vInfRhs;
644  afl.GetInfluenceFromVInfToPanel(vInfRhs);
645 
646  if (W.getParallel().myidWork == 0)
647  {
648 #pragma omp parallel for \
649  default(none) \
650  shared(locRhs, afl, bou, wakeRhs, vInfRhs, np) \
651  schedule(dynamic, DYN_SCHEDULE)
652  for (int i = 0; i < (int)afl.getNumberOfPanels(); ++i)
653  {
655  //Здесь учет скорости профиля и присоединенных слоев источников сделан только для прямолинейных панелей
656  locRhs(i) = - vInfRhs[i] - wakeRhs[i] + 0.25 * ((afl.getV(i) + afl.getV(i + 1)) & afl.tau[i]); //0.25 * (afl.getV(i) + afl.getV(i + 1))*afl.tau[i] - прямолинейные
657  if(W.getBoundary(bou).sheetDim > 1)
658  locRhs(np + i) = -vInfRhs[np + i] - wakeRhs[np + i];
659 
660  //влияние присоединенных слоев от самого себя и от других профилей
661  for (size_t q = 0; q < W.getNumberOfBoundary(); ++q)
662  {
663  const auto& sht = W.getBoundary(q).sheets;
664  const auto& iq = W.getIQ(bou, q);
665 
666  const Airfoil& aflOther = W.getAirfoil(q);
667  if (W.getMechanics(q).isDeform || W.getMechanics(q).isMoves)
668  {
669  for (size_t j = 0; j < aflOther.getNumberOfPanels(); ++j)
670  {
671  if ((i != j) || (bou != q))
672  {
673  locRhs(i) += -iq.first(i, j) * sht.attachedVortexSheet(j, 0);
674  locRhs(i) += -iq.second(i, j) * sht.attachedSourceSheet(j, 0); // getIQ(bou, q).second(i, j) пока забито нулем для криволинейных
675  }//if (i != j)
676  }//for j
677  }
678  }//for q
679  }//for i
680 
681  lastRhs[bou] = 0.0;
682 
683  for (size_t q = 0; q < afl.gammaThrough.size(); ++q)
684  lastRhs[bou] += afl.gammaThrough[q];
685 
686  //размазываем правую часть
687  for (size_t i = 0; i < nVars; ++i)
688  rhs(i + currentRow) = locRhs(i);
689 
690  rhs(currentRow + nVars) = lastRhs[bou];
691 
692  currentRow += nVars + 1;
693 
694  }// if (parallel.myidWork == 0)
695 
696  }// for bou
697 }
